import { z, defineCollection } from 'astro:content';

const foodsCollection = defineCollection({
  type: 'content',
  schema: z.object({
    name: z.string(),
    // We restrict the status to these 4 options so we don't make typos later
    status: z.enum(['Safe', 'Caution', 'Toxic', 'Emergency']),
    description: z.string(), // Short summary for search results
    keywords: z.array(z.string()).optional(), // Alt names (e.g. "Raisins")
  }),
});

export const collections = {
  'foods': foodsCollection,
};
